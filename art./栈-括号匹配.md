//思路：判断首位，末尾括号是不是合法的（比如第一个是 ] 就不合法）；两个栈首尾开始比较，看是不是对称的.懒得优化了

#include<iostream>

using namespace std;
const int Maxsize = 1e2 + 5;
typedef char DataType;

struct SqStack {
    DataType *base;
    DataType *top;
};

bool InitStack(SqStack &S) {
    S.base = new DataType[Maxsize];
    if (!S.base) {
        return false;
    }
    S.top = S.base;
    return true;
}

bool Push(SqStack &S, DataType e) {
    if (S.top - S.base == Maxsize) {
        return false;
    }
    *S.top++ = e;
    return true;
}

bool Pop(SqStack &S, DataType &e) {
    if (S.top == S.base) {
        return false;
    }
    e = *(--S.top);
    return true;
}

void Destroy(SqStack &S) {
    if (S.base || S.top) {
        delete S.base;
        delete S.top;
    }
}

int main() {
    int t;
    SqStack S1, S2;
    cin >> t;
    while (t--) {
        int sum1 = 0;//两个sum用来判断首尾的括号是合法的，不会出现第一个就是 ]、} 或者最后一个是 [、{
        int sum2 = 0;
        string s;
        cin >> s;
        InitStack(S1);
        InitStack(S2);
        int lens = s.size();
        for (int i = 0; i < lens; i++) {
            Push(S1, s[i]);
            Push(S2, s[lens - i - 1]);
        }
        DataType e1, e2;
        while (Pop(S1, e1)) {
            Pop(S2, e2);

            if (e1 == '{') {
                sum1--;
            } else if (e1 == '[') {
                sum2--;
            } else if (e1 == ']') {
                sum2++;
            } else if (e1 == '}') {
                sum1++;
            }

            if (sum1 < 0 || sum2 < 0) {
                break;
            }

            if (e2 == '[' && e1 != ']') {
                sum1 = -1;
                break;
            } else if (e1 == '[' && e2 != ']') {
                sum1 = -1;
                break;
            } else if (e1 == '{' && e2 != '}') {
                sum1 = -1;
                break;
            } else if (e2 == '{' && e1 != '}') {
                sum1 = -1;
                break;
            }
        }

        if (sum1 != 0 || sum2 != 0) {
            cout << "no" << endl;
        } else {
            cout << "yes" << endl;
        }
    }
    Destroy(S1);
    Destroy(S2);
    return 0;
}
