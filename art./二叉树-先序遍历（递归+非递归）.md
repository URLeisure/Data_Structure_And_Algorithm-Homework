// Description 二叉树-先序遍历（递归+非递归）
//用栈实现非递归，但是他之前说不让用 STL 就只能手写栈了
#include "iostream"
#include "cstring"

using namespace std;
#define Maxsize 10000

struct SqStack {
    int *top;
    int *base;
};

bool InitStack(SqStack &S) {
    S.top = new int[Maxsize];
    if (!S.top) {
        return false;
    }
    S.base = S.top;
    *S.top = *S.base = 0;
    return true;
}

bool Push(SqStack &S, int e) {
    if (S.top - S.base == Maxsize) {
        return false;
    }
    *S.top++ = e;
    return true;
}

int Pop(SqStack &S, int &e) {
    if (S.top == S.base) {
        return -1;
    }
    e = *--S.top;
    return true;
}

bool Empty(SqStack S) {
    if (S.top == S.base) {
        return false;
    }
    return true;
}

int Top(SqStack S) {
    if (S.top == S.base) {
        return -1;
    }
    return *(S.top - 1);
}

void Destroy(SqStack &S) {
    if (S.top || S.base) {
        delete[]S.top;
        delete[]S.base;
    }
}

int Tree[Maxsize];
SqStack S;

void PreOrder1(int n) {
    if (n) {
        cout << n << "\t";
        PreOrder1(Tree[2 * n]);
        PreOrder1(Tree[2 * n + 1]);
    }
}

void PreOrder2(int n) {
    if (n == 0) {
        return;
    }
    int e;
    Push(S, n);
    while (Empty(S)) {
        Pop(S, e);
        cout << e << "\t";
        int lchild = Tree[2 * e];
        int rchild = Tree[2 * e + 1];
        if (rchild) {
            Push(S, rchild);
        }
        if (lchild) {
            Push(S, lchild);
        }
    }
}

int main() {
    int n;
    while (cin >> n) {
        InitStack(S);
        memset(Tree, 0, sizeof Tree);
        for (int i = 1; i <= n; i++) {
            Tree[i] = i;
        }
        PreOrder1(Tree[1]);
        cout << endl;
        PreOrder2(Tree[1]);
        cout << endl;
        Destroy(S);
    }
    return 0;
}
