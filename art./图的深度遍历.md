#include<iostream>

using namespace std;
const int N = 1e3;

struct Edge {
    int to;
    int next;
} e[N];
int cnt, index;
int head[N], res[N], E[N][N];
bool vis[N];

void add(int u, int v) {
    e[cnt].to = v;
    e[cnt].next = head[u];
    head[u] = cnt++;
}

void init() {
    for (int i = 0; i < N; i++) {
        head[i] = -1;
        vis[i] = false;
    }
    cnt = index = 0;
}

void DFS(int u) {
    res[index++] = u;
    vis[u] = true;
    for (int i = head[u]; ~i; i = e[i].next) {
        int v = e[i].to;
        if (!vis[v]) {
            DFS(v);
        }
    }
}

void Print() {
    for (int i = 0; i < index; i++) {
        i == 0 || printf(",");
        cout << res[i];
    }
    cout << endl;
}

int main() {
    int t, n, flag;
    cin >> t;
    while (t--) {
        init();
        cin >> n;
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= n; j++) {
                cin >> E[i][j];
            }
        }
        //题目要求，有多个邻接点的话必须从小到大输出，但是链式前向星是反着的，我就反着存一遍，负负得正
        for (int i = n; i > 0; i--) {
            for (int j = n; j > 0; j--) {
                if (E[i][j] == 1) {
                    add(i, j);
                }
            }
        }

        DFS(1);
        Print();
    }
    return 0;
}
